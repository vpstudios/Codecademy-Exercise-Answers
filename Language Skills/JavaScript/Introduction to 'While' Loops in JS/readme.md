##8. When to 'while' and when to 'for'

```js
var count = 1;

while(count < 10) {
    count++;
    console.log("This is a for loop and the count is...");
    console.log("The count is " + count);
}
for(;count < 10;) {
   count++;
   console.log("This is a for loop and the count is...");
   console.log("The count is " + count);
}
```
